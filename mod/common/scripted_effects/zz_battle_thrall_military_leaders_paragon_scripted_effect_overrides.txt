# Modified:
# * nodes are immortal and have their age hidden (in create_leader instead of afterwards)
# * nodes have a flag corresponding to their intended council position
country_create_gestalt_councilor = {
	create_leader = {
		class = $CLASS$
		species = this
		name = $COUNCIL_POS$
		skill = 1
		immortal = yes
		hide_age = yes
		gender = indeterminable
		event_leader = yes
		skip_background_generation = yes
		can_manually_change_location = no
		can_see_in_list = no
		custom_description = $DESC$
		randomize_traits = no
		background_ethic = ethic_gestalt_consciousness
		effect = {
			change_leader_portrait = councilor_hive_$CLASS$_1
			set_council_position = $COUNCIL_POS$
			set_leader_flag = $COUNCIL_POS$
			set_leader_flag = leader_death_events_blocked
			if = {
				limit = { has_paragon_dlc = yes }
				switch = {
					trigger = leader_class
					admiral = {
						add_trait_no_notify = subclass_admiral_strategist
					}
					scientist = {
						add_trait_no_notify = subclass_scientist_researcher
					}
					general = {
						add_trait_no_notify = subclass_general_marshall
					}
					governor = {
						add_trait_no_notify = subclass_governor_visionary
					}
				}
			}
		}
	}
}
